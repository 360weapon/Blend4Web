"use strict";

/**
 * Gyroscope actions add-on.
 * Provides support for gyroscope.
 * @see http://www.html5rocks.com/en/tutorials/device/orientation/
 * @module gyroscope
 */

b4w.module["gyroscope"] = function(exports, require) {

var m_ctl   = require("controls");
var m_scenes     = require("scenes");
var m_cam        = require("camera");

var VERTICAL_ANGLE_THRESHOLD_UP = 75;
var VERTICAL_ANGLE_THRESHOLD_DOWN = -75;

exports.enable_camera_rotate = function() {
	var cam_obj = m_scenes.get_active_camera();
	create_gyro_sensors(cam_obj);
}

exports.disable_camera_rotate = function() {
	var cam_obj = m_scenes.get_active_camera();
	m_ctl.remove_sensor_manifold(cam_obj, "CAMERA_ROTATE_GYRO");
}

function create_gyro_sensors(obj) {

    var g_d_sensor = m_ctl.create_gyroscope_delta_angles_sensor();
    var g_a_sensor = m_ctl.create_gyroscope_angles_sensor();
    var logic_func = function(s) {
        return true;
    }

    var cam_rotate_cb = function(obj, id, pulse) {
        if (pulse > 0) {
            var delta_angles = m_ctl.get_sensor_payload(obj, id, 0);
            var angles = m_ctl.get_sensor_payload(obj, id, 1);
            if (angles[1] > VERTICAL_ANGLE_THRESHOLD_UP || angles[1] < VERTICAL_ANGLE_THRESHOLD_DOWN) {
                delta_angles[0] = 0;
                delta_angles[2] = 0;
            }
            var cam_type = m_cam.get_move_style(obj);
            switch (cam_type) {
            case m_cam.MS_TARGET_CONTROLS:
                m_cam.rotate_pivot(obj, delta_angles[0], delta_angles[1]);
                break;
            case m_cam.MS_EYE_CONTROLS:
                m_cam.rotate(obj, delta_angles[0], delta_angles[1]);
                break;
            case m_cam.MS_HOVER_CONTROLS:
                m_cam.rotate_hover_cam(obj, delta_angles[0]);
                break;
            }
        }
    }
    m_ctl.create_sensor_manifold(obj, "CAMERA_ROTATE_GYRO", 
            m_ctl.CT_CONTINUOUS, [g_d_sensor, g_a_sensor], logic_func, 
            cam_rotate_cb);
}

};
